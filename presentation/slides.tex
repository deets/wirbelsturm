\section{Die Rettung}

\frame{
  \frametitle{Der Ritter in schillernder Rüstung}
  Comet/Long-Poll\pause
  \begin{itemize}
    \item Client setzt AJAX-Request ab\pause
    \item Server antwortet erstmal nicht\pause
    \item Wenn Timout $\rightarrow$ neues Request\pause
    \item Wenn's was zu sagen gibt $\rightarrow$ Server antwortet
  \end{itemize}
}

\frame{
  \frametitle{Rostige Rüstungsteile}
  \begin{itemize}
    \item Lange offene Requests belegen Resourcen\pause
    \item ``Klassische'' Webserver wie Apache/CherryPy/Paster haben damit Probleme\pause
    \item Der SuperChat skaliert nicht!
  \end{itemize}
}
